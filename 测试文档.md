### Go代码

```go
// 读取剪贴板数据
func (s *systemService) ReadClipboard() (resp types.JSResp) {
	resp.Success = true
	// 初始化剪贴板
	clipboard.Init()

	// 从剪贴板读取文本数据
	data := clipboard.Read(clipboard.FmtText)
	if data == nil {
		resp.Success = true
	}
	// 将字节切片转换成字符串
	text := string(data)
	resp.Data = text
	return
}
```

### Shell代码

```bash
#!/bin/bash
__main() {
    last_update_file=~/.xterminal/last_update
    if [[ -f $last_update_file ]]; then
        last_update=$(cat $last_update_file)
        now=$(date +%s)
        if [[ $((now - last_update)) -lt 2 ]]; then
            exit
        fi
    fi
    output_file=~/.xterminal/$(date +%s%N).stats
    for param in "$@"
    do
        dir=~/.xterminal/sub/$param
        file=$dir/$param.sh
        cd "$dir" || exit
        chmod +x "$file"
        ./"$param".sh
        cat "$dir/$param".stats >> "$output_file"
    done
    mv "$output_file" ~/.xterminal/output.stats
    echo $(date +%s) > $last_update_file
}
__main "$@"
```
